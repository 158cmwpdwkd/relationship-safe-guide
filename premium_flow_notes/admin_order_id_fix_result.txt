3. 현재 sessions 데이터의 조회 방식 요약
- admin_home은 asyncpg pool로 raw SQL을 실행한다.
- 기존 조회 SQL은 user_sessions u
  LEFT JOIN reports r ON u.sid = r.sid
  만 사용했다.
- 템플릿에는 sessions = [dict(r) for r in rows] 로 전달한다.
- 즉, 템플릿의 각 row는 SQL SELECT에 포함된 컬럼만 접근 가능하다.

4. order_id가 안 뜨는 원인 분석
- templates/admin.html에는 이미 {{ s.order_id }} 컬럼이 들어 있었다.
- 하지만 admin_home의 SELECT 문에는 o.order_id 또는 orders 조인이 없었다.
- 따라서 sessions 각 row dict 안에 order_id 키가 생성되지 않았다.
- 결과적으로 템플릿에서 {{ s.order_id }} 는 비어 보이거나 표시되지 않았다.

5. 실제로 어느 모델/테이블에 order_id가 있는지
- 실제 코드 기준 모델: app/models.py
- 모델: Order
- 테이블: orders
- 컬럼: order_id
- 관계 기준:
  - Order.sid -> user_sessions.sid
  - Report.sid -> user_sessions.sid
- 즉 admin 목록에서 order_id를 보이게 하려면 orders 테이블을 sid 기준으로 조인해야 한다.

8. admin 라우트 파일의 최종 수정 코드 전체

```python
import os
import asyncpg
import secrets
from datetime import datetime, timedelta, timezone

from fastapi import APIRouter, Request, Depends, HTTPException
from fastapi.responses import HTMLResponse, RedirectResponse
from fastapi.templating import Jinja2Templates
from fastapi.security import HTTPBasic, HTTPBasicCredentials

router = APIRouter(prefix="/_admin", tags=["admin"])
templates = Jinja2Templates(directory="templates")

security = HTTPBasic()


def require_admin(credentials: HTTPBasicCredentials = Depends(security)):
    admin_user = os.getenv("ADMIN_USER", "admin")
    admin_pass = os.getenv("ADMIN_PASS", "admin")

    correct_user = secrets.compare_digest(credentials.username, admin_user)
    correct_pass = secrets.compare_digest(credentials.password, admin_pass)

    if not (correct_user and correct_pass):
        raise HTTPException(status_code=401, headers={"WWW-Authenticate": "Basic"})


async def get_pool(request: Request) -> asyncpg.Pool:
    pool = request.app.state.db_pool
    return pool


@router.get("", response_class=HTMLResponse)
async def admin_home(
    request: Request,
    _: HTTPBasicCredentials = Depends(require_admin),
    pool: asyncpg.Pool = Depends(get_pool)
):

    sql = """
    SELECT
    u.sid,
    u.phone,
    u.email,
    u.risk_level,
    u.created_at,
    r.report_token,
    o.order_id,
    COALESCE(r.status, o.status) AS status
    FROM user_sessions u
    LEFT JOIN reports r ON u.sid = r.sid
    LEFT JOIN orders o ON u.sid = o.sid
    ORDER BY u.created_at DESC
    LIMIT 50
    """

    async with pool.acquire() as conn:
        rows = await conn.fetch(sql)

    sessions = [dict(r) for r in rows]

    return templates.TemplateResponse(
        "admin.html",
        {
            "request": request,
            "sessions": sessions
        }
    )


@router.get("/session/{sid}", response_class=HTMLResponse)
async def session_detail(
    request: Request,
    sid: str,
    _: HTTPBasicCredentials = Depends(require_admin),
    pool: asyncpg.Pool = Depends(get_pool)
):

    async with pool.acquire() as conn:

        session = await conn.fetchrow("""
        SELECT * FROM user_sessions
        WHERE sid = $1
        """, sid)

        report = await conn.fetchrow("""
        SELECT * FROM reports
        WHERE sid = $1
        """, sid)

        order = await conn.fetchrow("""
        SELECT * FROM orders
        WHERE sid = $1
        ORDER BY paid_at DESC NULLS LAST, order_id DESC
        LIMIT 1
        """, sid)

    if not session:
        raise HTTPException(404)

    return templates.TemplateResponse(
        "session_detail.html",
        {
            "request": request,
            "session": dict(session),
            "report": dict(report) if report else None,
            "order": dict(order) if order else None,
        }
    )


@router.post("/delete_recent_hour")
async def delete_recent_hour(
    _: HTTPBasicCredentials = Depends(require_admin),
    pool: asyncpg.Pool = Depends(get_pool),
):
    async with pool.acquire() as conn:
        async with conn.transaction():
            await conn.execute("""
                DELETE FROM reports
                WHERE sid IN (
                    SELECT sid FROM user_sessions
                    WHERE created_at >= (NOW() AT TIME ZONE 'UTC') - INTERVAL '1 hour'
                )
            """)
            await conn.execute("""
                DELETE FROM user_sessions
                WHERE created_at >= (NOW() AT TIME ZONE 'UTC') - INTERVAL '1 hour'
            """)

    return RedirectResponse("/_admin", status_code=303)


@router.post("/delete_today")
async def delete_today(
    _: HTTPBasicCredentials = Depends(require_admin),
    pool: asyncpg.Pool = Depends(get_pool),
):
    async with pool.acquire() as conn:
        async with conn.transaction():
            await conn.execute("""
                DELETE FROM reports
                WHERE sid IN (
                    SELECT sid FROM user_sessions
                    WHERE created_at >= DATE_TRUNC('day', NOW() AT TIME ZONE 'UTC')
                      AND created_at <  DATE_TRUNC('day', NOW() AT TIME ZONE 'UTC') + INTERVAL '1 day'
                )
            """)
            await conn.execute("""
                DELETE FROM user_sessions
                WHERE created_at >= DATE_TRUNC('day', NOW() AT TIME ZONE 'UTC')
                  AND created_at <  DATE_TRUNC('day', NOW() AT TIME ZONE 'UTC') + INTERVAL '1 day'
            """)

    return RedirectResponse("/_admin", status_code=303)


@router.post("/delete_all")
async def delete_all(
    _: HTTPBasicCredentials = Depends(require_admin),
    pool: asyncpg.Pool = Depends(get_pool),
):
    async with pool.acquire() as conn:
        async with conn.transaction():
            async def del_if_exists(table: str):
                exists = await conn.fetchval("SELECT to_regclass($1)", f"public.{table}")
                if exists:
                    await conn.execute(f"DELETE FROM {table}")

            await del_if_exists("message_schedules")
            await del_if_exists("paid_surveys")
            await del_if_exists("orders")
            await del_if_exists("reports")
            await del_if_exists("user_sessions")

    return RedirectResponse("/_admin", status_code=303)
```
