1. 유지할 설계

- 기존 무료 리포트 구조 유지
  - 아임웹 wrapper
  - iframe
  - 백엔드 HTML 문자열 반환
- 외부 진입 키는 token 중심 유지
- premium 상태 판정은 기존 `resolve_premium_state()` 사용
- premium 본문 HTML은 백엔드 renderer가 생성
- 아임웹은 wrapper 역할만 수행

2. 수정된 설계

- `routes_report.py`
  - 하나의 DB 세션만 사용
  - `resolve_premium_state()`를 먼저 호출
  - INVALID 상태를 먼저 처리
  - premium token이면 free fallback 금지
  - `/r/{token}`는 wrapper가 아닌 direct access 시에도 premium 상태 페이지를 안전하게 반환

- `premium_renderer.py`
  - premium 본문 HTML shell 정리
  - processing / blocked / need-survey / invalid 상태용 HTML 렌더러 추가
  - iframe height postMessage 포함

- `imweb/premium_report.html`
  - 새 wrapper 파일 생성
  - token 읽기: query / path / localStorage
  - entry API 호출
  - wrapper가 상태 UI 담당
  - READY일 때만 iframe 표시
  - PROCESSING은 wrapper loading + polling

3. fallback 정책

- free fallback 허용 조건
  - `reports.report_token`은 존재하지만
  - 해당 sid에 `orders` 없음
  - `paid_surveys` 없음
  - premium html/markdown 없음
  - 즉 “순수 무료 토큰”인 경우만 free HTML 반환

- premium token fallback 정책
  - premium flow 흔적이 있으면 free HTML 절대 반환 금지
  - premium flow 흔적 판단:
    - `orders.sid` 존재 또는
    - `paid_surveys.sid` 존재 또는
    - `report.html/report.markdown` 존재

- 최종 분기
  - READY -> premium html
  - PROCESSING -> processing html
  - NOT_PAID -> premium blocked html
  - NEED_SURVEY -> premium need-survey html
  - INVALID -> invalid html
  - 순수 무료 토큰만 free html

4. wrapper vs iframe 역할 분리

- wrapper가 담당
  - token 읽기
  - entry API 호출
  - NOT_PAID UI
  - NEED_SURVEY UI
  - PROCESSING loading UI + polling
  - READY 시 iframe 표시
  - iframe height 반영

- iframe이 담당
  - premium 본문 HTML 표시
  - direct access 시 PROCESSING / INVALID / NOT_PAID / NEED_SURVEY 보호용 HTML 반환
  - wrapper가 READY 전에 iframe을 열지 않으므로 정상 사용자 UX에서는 wrapper와 중복되지 않음

```python
from fastapi import APIRouter, HTTPException, Query
from fastapi.responses import HTMLResponse
from sqlalchemy.orm import Session

from .db import SessionLocal
from .models import Order, PaidSurvey, Report, UserSession
from .report import make_report_html
from .services.premium_report import resolve_premium_state
from .services.reporting.premium_renderer import (
    render_premium_processing_html,
    render_premium_state_html,
)

router = APIRouter()


def load_report_and_session_by_token(token: str, db: Session):
    rep = db.query(Report).filter(Report.report_token == token).first()
    if not rep:
        raise HTTPException(status_code=404, detail="Invalid token")

    sess = db.query(UserSession).filter(UserSession.sid == rep.sid).first()
    if not sess:
        raise HTTPException(status_code=404, detail="Session not found")

    return rep, sess


def _is_premium_flow_token(*, report: Report, db: Session) -> bool:
    has_order = db.query(Order).filter(Order.sid == report.sid).first() is not None
    has_paid_survey = db.query(PaidSurvey).filter(PaidSurvey.sid == report.sid).first() is not None
    has_premium_content = bool((report.html or "").strip() or (report.markdown or "").strip())
    return has_order or has_paid_survey or has_premium_content


def resolve_report_html(token: str) -> HTMLResponse:
    db = SessionLocal()
    try:
        state = resolve_premium_state(report_token=token, db=db)
        if state.state == "INVALID_REPORT_TOKEN":
            return HTMLResponse(
                render_premium_state_html(
                    state="INVALID_REPORT_TOKEN",
                    title="유효하지 않은 리포트 링크입니다",
                    message=state.user_message,
                    cta_label="홈으로 이동",
                    cta_href="/",
                ),
                status_code=404,
            )

        report, session = load_report_and_session_by_token(token, db)
        is_premium_flow = _is_premium_flow_token(report=report, db=db)

        if state.state == "READY" and (report.html or "").strip():
            return HTMLResponse(report.html)

        if is_premium_flow:
            if state.state == "PROCESSING":
                return HTMLResponse(
                    render_premium_processing_html(message=state.user_message),
                    status_code=202,
                )

            if state.state == "NOT_PAID":
                return HTMLResponse(
                    render_premium_state_html(
                        state="NOT_PAID",
                        title="결제가 필요합니다",
                        message=state.user_message,
                    ),
                    status_code=403,
                )

            if state.state == "NEED_SURVEY":
                return HTMLResponse(
                    render_premium_state_html(
                        state="NEED_SURVEY",
                        title="유료 설문 작성이 필요합니다",
                        message=state.user_message,
                    ),
                    status_code=409,
                )

            return HTMLResponse(
                render_premium_processing_html(message=state.user_message),
                status_code=202,
            )

        return HTMLResponse(
            make_report_html(
                risk_level=session.risk_level,
                impulse=session.impulse_index,
                fear_type=session.fear_type,
            )
        )
    finally:
        db.close()


@router.get("/r/{token}", response_class=HTMLResponse)
async def view_report(token: str):
    return resolve_report_html(token)


@router.get("/report", response_class=HTMLResponse)
async def view_report_query(token: str = Query(...)):
    return resolve_report_html(token)
```

6. premium renderer 구조

- 파일: `app/services/reporting/premium_renderer.py`
- 핵심 함수
  - `render_premium_report_html(markdown_text, metrics=None)`
  - `render_premium_processing_html(message=...)`
  - `render_premium_state_html(state, title, message, cta_label=None, cta_href=None)`

- 구조
  - `_shell()` 에서 공통 CSS + postMessage height 처리
  - READY 상태는 hero + metrics + markdown body + help/footer
  - PROCESSING/NOT_PAID/NEED_SURVEY/INVALID 는 state-card 템플릿 사용

- 역할
  - routes_report가 상태만 정하고
  - renderer가 최종 HTML 문자열을 생성한다

```html
    const API_BASE = "https://relationship-safe-guide-api.onrender.com";
    const ENTRY_URL = API_BASE + "/api/premium/entry";
    const STORAGE_KEY = "rl_report_token";
    const POLL_INTERVAL_MS = 3000;

    function getQueryToken() {
      try {
        return new URLSearchParams(window.location.search).get("token") || "";
      } catch (e) {
        return "";
      }
    }

    function getPathToken() {
      try {
        const match = (window.location.pathname || "").match(/\/r\/([^/?#]+)/);
        return match ? decodeURIComponent(match[1]) : "";
      } catch (e) {
        return "";
      }
    }

    function getStoredToken() {
      try {
        return localStorage.getItem(STORAGE_KEY) || "";
      } catch (e) {
        return "";
      }
    }

    function setStoredToken(token) {
      try {
        if (token) localStorage.setItem(STORAGE_KEY, token);

    function getToken() {
      return getQueryToken() || getPathToken() || getStoredToken() || "";

    async function fetchEntry(token) {
      const url = new URL(ENTRY_URL);
      url.searchParams.set("token", token);
      const res = await fetch(url.toString(), { method: "GET" });
      if (!res.ok) {
        throw new Error("ENTRY_FETCH_FAILED");
      }
      return res.json();
    }

    function clearPolling() {
      if (pollHandle) {
        clearTimeout(pollHandle);
        pollHandle = null;
      }
    }

    function scheduleRecheck(token) {
      clearPolling();
      pollHandle = setTimeout(function () {
        boot(token);
      }, POLL_INTERVAL_MS);
    }

    async function boot(initialToken) {
      const token = initialToken || getToken();
      if (!token) {
        showError();
        return;
      }

      setStoredToken(token);
      showLoading("프리미엄 리포트를 확인하는 중입니다", "잠시만 기다려 주세요.");

      try {
        const data = await fetchEntry(token);
        const state = data.state || "";

        if (state === "INVALID_REPORT_TOKEN") {
          clearPolling();
          showError();
          return;
        }

        if (state === "NOT_PAID") {
          clearPolling();
          showState(
            "결제가 필요합니다",
            data.user_message || "프리미엄 리포트를 보려면 결제를 완료해 주세요.",
            "결제 진행하기",
            data.next_url || ("/premium?token=" + encodeURIComponent(token))
          );
          return;
        }

        if (state === "NEED_SURVEY") {
          clearPolling();
          showState(
            "유료 설문 작성이 필요합니다",
            data.user_message || "설문 작성 후 프리미엄 리포트가 생성됩니다.",
            "설문 작성하기",
            data.next_url || "#"
          );
          return;
        }

        if (state === "PROCESSING") {
          showLoading(
            "프리미엄 리포트를 생성 중입니다",
            data.user_message || "생성이 완료되면 자동으로 표시됩니다."
          );
          scheduleRecheck(token);
          return;
        }

        if (state === "READY") {
          clearPolling();
          showFrame(token);
          return;
        }

        clearPolling();
        showError();
      } catch (e) {
        clearPolling();
        showError();
      }
    }

    frameEl.addEventListener("load", function () {
      frameEl.classList.add("loaded");

    window.addEventListener("message", function (event) {

      if (

      if (Math.abs(target - lastHeight) <= 8) return;

      frameEl.style.height = target + "px";
      lastHeight = target;
    });

    boot("");
```
